17/// Android client integration (Kotlin).
18/// 
19/// The Android client connects via JNI and requires the pre-generated Protobuf classes.

21/// [**Download Android Starter ZIP**](../android_client.zip)
22/// 
23/// ```kotlin
24/// // c:\Users\arjun\importer\rust_core\src\client\android\connect.kt
25/// 
26/// package com.zoho.arattai.client
27/// 
28/// import android.util.Log
29/// import com.zoho.arattai.core.messages.WhatsAppExport // Pre-generated using protoc
30/// 
31/// class WhatsAppAndroidConnector {
32///     init {
33///         System.loadLibrary("rust_core")
34///     }
35/// 
36///     private external fun parseChatNative(path: String): ByteArray?
37/// 
38///     fun handleChatImport(zipPath: String) {
39///         val protoBytes = parseChatNative(zipPath)
40///         if (protoBytes != null) {
41///             val export = WhatsAppExport.parseFrom(protoBytes)
42///             Log.i("WhatsAppParser", "Imported: ${export.chatName}")
43///         }
44///     }
45/// }

49/// iOS client integration (Swift).
50/// 
51/// The iOS client connects via a static C-FFI bridge and requires SwiftProtobuf.
52/// 
53/// [**Download iOS Starter ZIP**](../ios_client.zip)
54/// 
55/// ```swift
56/// // c:\Users\arjun\importer\rust_core\src\client\ios\connect.swift
57/// 
58/// import Foundation
59/// 
60/// class WhatsAppiOSImporter {
61///     func startImport(fromPath path: String) {
62///         let cPath = (path as NSString).utf8String
63///         let buffer = parse_chat_ffi(cPath)
64///         defer { free_byte_buffer(buffer) }
65/// 
66///         if let rawData = buffer.data, buffer.len > 0 {
67///             let binaryData = Data(bytes: rawData, count: Int(buffer.len))
68///             let export = try Whatsapp_WhatsAppExport(serializedData: binaryData)
69///             print("Imported: \(export.chatName)")
70///         }
71///     }
72/// }
73/// ```
74pub mod ios {}
