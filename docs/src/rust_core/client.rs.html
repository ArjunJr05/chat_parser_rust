1//! This module contains documentation and bridge code for the various
2//! platform-specific clients that connect to the Rust core.
3//! 
4//! # Integration Examples
5//! 
6//! Because the client code lives in non-Rust files (.kt, .swift, .vue), 
7//! the source code is provided inside the documentation blocks below.
8//! 
9//! # Downloads
10//! 
11//! Download the complete ready-to-use starter package for your platform.
12//! Each ZIP contains **everything** needed — the connector code, the compiled
13//! Rust binary, and the Protobuf schema.
14//! 
15//! | Platform | Download | Contains |
16//! |----------|----------|----------|
17//! | Android | [Download android_client.zip](../android_client.zip) | `connect.kt` + `librust_core.so` (arm64, armv7, x86_64) + `messages.proto` |
18//! | iOS | [Download ios_client.zip](../ios_client.zip) | `connect.swift` + `rust_core.h` + `librust_core.a` + `messages.proto` |
19//! | Web | [Download web_client.zip](../web_client.zip) | `connect.vue` + `wasm-init.js` + `rust_core_bg.wasm` + `rust_core.js` + `messages.proto` |
20//!
21//! ## How to use the files
22//!
23//! ### Android (`.so` file)
24//! 1. Unzip and place the `jniLibs/` folder into your Android project's `app/src/main/` directory.
25//! 2. Place `connect.kt` in your Kotlin source folder.
26//! 3. Run `protoc` on `messages.proto` to generate the Java Protobuf classes.
27//!
28//! ### iOS (`.a` file)
29//! 1. Drag `librust_core.a` into your Xcode project → Link Binary with Libraries.
30//! 2. Add `rust_core.h` to your **Bridging Header**.
31//! 3. Run `protoc --swift_out=.` on `messages.proto` to generate Swift Protobuf classes.
32//!
33//! ### Web (`.wasm` file)
34//! 1. Place `rust_core_bg.wasm` and `rust_core.js` in your project's `public/wasm/` folder.
35//! 2. Place `messages.proto` in your `public/proto/` folder.
36//! 3. Install dependencies: `npm install vite vite-plugin-wasm protobufjs`
37//! 4. Use `connect.vue` and `wasm-init.js` as the entry point.
38
39/// Android client integration (Kotlin).
40/// 
41/// The Android client connects via JNI using the `.so` shared library.
42/// 
43/// [**Download Android Starter ZIP**](../android_client.zip)
44/// 
45/// **ZIP Contents:**
46/// ```text
47/// android_client.zip
48/// ├── connect.kt                          ← Copy to your Kotlin source folder
49/// ├── messages.proto                      ← Run protoc to generate Java classes
50/// └── jniLibs/
51///     ├── arm64-v8a/librust_core.so       ← Modern phones (most common)
52///     ├── armeabi-v7a/librust_core.so     ← Older 32-bit phones
53///     └── x86_64/librust_core.so          ← Emulators
54/// ```
55/// 
56/// ### How to connect (Kotlin):
57/// ```kotlin
58/// package com.zoho.arattai.client
59/// 
60/// import android.util.Log
61/// import com.zoho.arattai.core.messages.WhatsAppExport // Pre-generated using protoc
62/// 
63/// class WhatsAppAndroidConnector {
64///     init {
65///         // Loads jniLibs/arm64-v8a/librust_core.so automatically
66///         System.loadLibrary("rust_core")
67///     }
68/// 
69///     private external fun parseChatNative(path: String): ByteArray?
70/// 
71///     fun handleChatImport(zipPath: String) {
72///         val protoBytes = parseChatNative(zipPath)
73///         if (protoBytes != null) {
74///             val export = WhatsAppExport.parseFrom(protoBytes)
75///             Log.i("WhatsAppParser", "Imported: ${export.chatName}")
76///             for (msg in export.messagesList) {
77///                 if (msg.hasText()) Log.d("Msg", msg.text.text)
78///             }
79///         }
80///     }
81/// }
82/// ```
83pub mod android {}

85/// iOS client integration (Swift).
86/// 
87/// The iOS client connects via a C-FFI bridge using the `.a` static library.
88/// 
89/// [**Download iOS Starter ZIP**](../ios_client.zip)
90/// 
91/// **ZIP Contents:**
92/// ```text
93/// ios_client.zip
94/// ├── connect.swift      ← Add to your Xcode Swift project
95/// ├── rust_core.h        ← Add to your Xcode Bridging Header
96/// ├── librust_core.a     ← Drag into Xcode → Link Binary With Libraries
97/// └── messages.proto     ← Run: protoc --swift_out=. messages.proto
98/// ```
99/// 
100/// ### How to connect (Swift):
101/// ```swift
102/// import Foundation
103/// 
104/// class WhatsAppiOSImporter {
105///     func startImport(fromPath path: String) {
106///         let cPath = (path as NSString).utf8String
107///         // parse_chat_ffi is exposed from librust_core.a via rust_core.h
108///         let buffer = parse_chat_ffi(cPath)
109///         defer { free_byte_buffer(buffer) }
110/// 
111///         if let rawData = buffer.data, buffer.len > 0 {
112///             let binaryData = Data(bytes: rawData, count: Int(buffer.len))
113///             let export = try Whatsapp_WhatsAppExport(serializedData: binaryData)
114///             print("Imported: \(export.chatName)")
115///             for msg in export.messages where msg.hasText {
116///                 print("[\(msg.text.base.sender)]: \(msg.text.text_p)")
117///             }
118///         }
119///     }
120/// }
121/// ```
122pub mod ios {}

124/// Web client integration (Vue + WASM).
125/// 
126/// The Web client loads the Rust engine in the browser using the `.wasm` file.
127/// 
128/// [**Download Web Starter ZIP**](../web_client.zip)
129/// 
130/// **ZIP Contents:**
131/// ```text
132/// web_client.zip
133/// ├── connect.vue          ← The main Vue component (UI + logic)
134/// ├── wasm-init.js         ← Loads WASM and Protobuf (the bridge)
135/// ├── vite.config.js       ← Required Vite config for WASM support
136/// ├── rust_core_bg.wasm    ← The compiled Rust engine (put in public/wasm/)
137/// ├── rust_core.js         ← JS glue for WASM (put in public/wasm/)
138/// └── messages.proto       ← Protobuf schema (put in public/proto/)
139/// ```
140/// 
141/// ### How to connect (JavaScript):
142/// ```javascript
143/// // wasm-init.js — loads the .wasm engine and the .proto schema
144/// import init, { parse_chat_wasm } from './rust_core.js';
145/// import protobuf from 'protobufjs';
146/// 
147/// let root = null;
148/// 
149/// export async function initWasm() {
150///     await init(); // Loads rust_core_bg.wasm
151///     root = await protobuf.load('/proto/messages.proto');
152/// }
153/// 
154/// export function decodeWhatsAppExport(bytes) {
155///     const WhatsAppExport = root.lookupType("whatsapp.WhatsAppExport");
156///     return WhatsAppExport.toObject(WhatsAppExport.decode(bytes), {
157///         longs: String, enums: String, defaults: true, arrays: true
158///     });
159/// }
160/// 
161/// export { parse_chat_wasm };
162/// ```
163pub mod web {}
