18#[unsafe(no_mangle)]
19pub extern "C" fn parse_chat_ffi(path: *const c_char) -> ByteBuffer {
20    if path.is_null() { return ByteBuffer { data: std::ptr::null_mut(), len: 0 }; }
21    let c_str = unsafe { CStr::from_ptr(path) };
22    let path_str = match c_str.to_str() { Ok(s) => s, Err(_) => return ByteBuffer { data: std::ptr::null_mut(), len: 0 } };
23    match WhatsAppChatParser::parse(path_str) {
24        Ok(export) => {
25            let proto_export = ProtoExport { chat_name: export.get_chat_name().to_string(), messages: export.get_all_messages().to_vec() };
26            let mut buf = Vec::new();
27            if proto_export.encode(&mut buf).is_ok() {
28                let len = buf.len();
29                let data = Box::into_raw(buf.into_boxed_slice()) as *mut u8;
30                ByteBuffer { data, len }
31            } else { ByteBuffer { data: std::ptr::null_mut(), len: 0 } }
32        }
33        Err(_) => ByteBuffer { data: std::ptr::null_mut(), len: 0 },
34    }
35}

37#[cfg(not(target_arch = "wasm32"))]
38#[unsafe(no_mangle)]
39pub extern "C" fn free_byte_buffer(buffer: ByteBuffer) {
40    if !buffer.data.is_null() { unsafe { let _ = Box::from_raw(std::slice::from_raw_parts_mut(buffer.data, buffer.len)); } }
41}
